class Vector2d:
    typecode = 'd'

    def __iter__(self): # x, y = my_vector
        return (i for i in (self.x, self.y))

    def __repr__(self):
        class_name = type(self).__name__
        return '{}({!r}, {!r})'.format(class_name, *self)

    def __bytes__(self):
        return (bytes([ord(self.typecode)]) +

    @classmethod
