  const int len  = sizeof(tri) / sizeof(tri[0]);
  const int base = (sqrt(8*len + 1) - 1) / 2;
  int step       = base - 1;
  int stepc      = 0;

  int i;
  for (i = len - base - 1; i >= 0; --i) {
    tri[i] += max(tri[i + step], tri[i + step + 1]);
    if (++stepc == step) {
      step--;
      stepc = 0;
    }
  }

  printf("%d\n", tri[0]);
  return 0;
}
